
CREATE TABLE X_PH_DML_AUDIT
(
  ID                NUMBER                      NOT NULL,
  DATE_TIME_CHANGE  DATE,
  TABLE_NAME        VARCHAR2(30 BYTE),
  OPERATION         VARCHAR2(3 BYTE),
  DB_USER           VARCHAR2(20 BYTE),
  OS_USER           VARCHAR2(20 BYTE)
);

CREATE INDEX X_PH_DML_AUDIT_DATE_I ON X_PH_DML_AUDIT
(DATE_TIME_CHANGE);

CREATE UNIQUE INDEX X_PH_DML_AUDIT_PK ON X_PH_DML_AUDIT
(ID);


CREATE INDEX X_PH_DML_TABLE_I ON X_PH_DML_AUDIT
(TABLE_NAME);

GRANT SELECT ON X_PH_DML_AUDIT TO MANAGE_APPLICATION;

CREATE SEQUENCE X_PH_DML_AUDIT_SEQ;
